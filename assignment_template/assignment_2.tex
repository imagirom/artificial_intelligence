% Header and footer for when a page split occurs within a problem environment
\newcommand{\enterQuestionHeader}[1]{
\nobreak\extramarks{#1}{#1 continued on next page\ldots}\nobreak
\nobreak\extramarks{#1 (continued)}{#1 continued on next page\ldots}\nobreak
}

% Header and footer for when a page split occurs between problem environments
\newcommand{\exitQuestionHeader}[1]{
\nobreak\extramarks{#1 (continued)}{#1 continued on next page\ldots}\nobreak
\nobreak\extramarks{#1}{}\nobreak
}

\setcounter{secnumdepth}{0} % Removes default section numbers
\newcounter{questionCounter} % Creates a counter to keep track of the number of problems

\newcommand{\questionName}{}
\newenvironment{question}[1][Question \arabic{questionCounter}]{ % Makes a new environment called question which takes 1 argument (custom name) but the default is "Question #"
\stepcounter{questionCounter} % Increase counter for number of problems
\renewcommand{\questionName}{#1} % Assign \questionName the name of the problem
\section{\questionName} % Make a section in the document with the custom problem count
\enterQuestionHeader{\questionName} % Header and footer within the environment
}{
\exitQuestionHeader{\questionName} % Header and footer after the environment
}

\newcommand{\problemAnswer}[1]{ % Defines the problem answer command with the content as the only argument
\noindent\framebox[\columnwidth][c]{\begin{minipage}{0.98\columnwidth}#1\end{minipage}} % Makes the box around the problem answer and puts the content inside
}

\newcommand{\assignmentSectionName}{}
\newenvironment{assignmentSection}[1]{ % New environment for sections within assignment problems, takes 1 argument - the name of the section
\renewcommand{\assignmentSectionName}{#1} % Assign \assignmentSectionName to the name of the section from the environment argument
\subsection{\assignmentSectionName} % Make a subsection with the custom name of the subsection
\enterQuestionHeader{\questionName\ [\assignmentSectionName]} % Header and footer within the environment
}{
\enterQuestionHeader{\questionName} % Header and footer after the environment
}
